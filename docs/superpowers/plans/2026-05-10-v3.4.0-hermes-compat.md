# V3.4.0 Hermes Compatibility Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship V3.4.0 with explicit Hermes compatibility strategies, reliable multi-profile routing, per-bot card titles, cron final-result cards, attachment summaries, card reply context, and issue #23 regression coverage.

**Architecture:** Keep sidecar-only. Hermes gets version-specific minimal marker hooks; all stateful behavior stays in `hermes_feishu_card/`. Detection chooses `legacy_gateway_run` or `gateway_run_013_plus` using version plus AST anchor capabilities.

**Tech Stack:** Python 3.9+, `aiohttp`, `PyYAML`, `pytest`, `pytest-asyncio`, AST-based patching.

---

```python
def _profile_identity(local_vars: dict[str, Any], source_obj: Any, message_obj: Any) -> tuple[str, str]:
    env_profile = os.environ.get("HERMES_FEISHU_CARD_PROFILE_ID", "").strip()
    if env_profile:
        return env_profile, "env"
    direct = (
        _first_string(local_vars, ("profile_id", "hermes_profile", "profile"))
        or _first_attr_string(source_obj, ("profile_id", "hermes_profile", "profile"))
        or _first_attr_string(message_obj, ("profile_id", "hermes_profile", "profile"))
    )
    if direct:
        return direct, "locals"
    hermes_home = os.environ.get("HERMES_HOME", "").strip()
    profile = _profile_from_path(hermes_home)
    if profile:
        return profile, "hermes_home"
    return "default", "fallback_default"


def _profile_from_path(path: str) -> str | None:
    if not path:
        return None
    parts = Path(path).expanduser().parts
    if "profiles" in parts:
        index = parts.index("profiles")
        if index + 1 < len(parts):
            candidate = parts[index + 1].strip()
            if candidate:
                return candidate
    return None
```

```python
MEDIA_RE = re.compile(r"MEDIA:([^\\s\\]]+)")
LOCAL_FILE_RE = re.compile(r"(?<![\\w/])(/[^\\s`]+\\.(?:png|jpg|jpeg|webp|gif|pdf|txt|md|csv|xlsx|docx|mp3|wav|ogg|mp4|mov|webm))")

def _extract_attachments(text: str) -> list[dict[str, str]]:
    seen = set()
    out = []
    for raw in list(MEDIA_RE.findall(text or "")) + list(LOCAL_FILE_RE.findall(text or "")):
        name = Path(raw).name.strip()
        if not name or name in seen:
            continue
        seen.add(name)
        out.append({"kind": _attachment_kind(name), "name": name, "summary": name})
    return out

def _attachment_kind(name: str) -> str:
    ext = Path(name).suffix.lower()
    if ext in {".png", ".jpg", ".jpeg", ".webp", ".gif"}:
        return "image"
    if ext in {".mp3", ".wav", ".ogg"}:
        return "audio"
    if ext in {".mp4", ".mov", ".webm"}:
        return "video"
    if ext:
        return "file"
    return "unknown"
```
